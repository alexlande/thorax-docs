Getting Started

$ ->
ExampleView = Thorax.View.extend
  name: '$-example'
  cats: Thorax.Collections.Cats.instance
  events:
    'click li': (event) ->
      name = $(event.target).model().get('name')
      $('body').append 'clicked on ' + name + '<br>'
  template: """
    <h3>$ Example</h3>
    {{#collection cats tag="ul"}}
      <li>{{name}}</li>
    {{/collection}}
  """

Application.setView new ExampleView​​​​​

$ ->
  
  mittens = Thorax.Collections.Cats.instance.at(0)
  TodosView = Thorax.View.extend
    name: 'todos'
    title: mittens.get('name') + " Todos"
    todos: mittens.get('todos')
    events:
      'change input[type="checkbox"]': (event) ->
        target = $(event.target)
        target.model().set done: !!target.attr('checked')
      'submit form': (event) ->
        event.preventDefault()
        input = @$('input[type="text"]')
        @todos.add item: input.val()
        input.val('')
    template: """
      <h3>{{title}}</h3>
      {{#collection todos tag="ul"}}
        <li{{#if done}} class="done"{{/if}}>
          <input type="checkbox" name="done"{{#if done}} checked="checked"{{/if}}>
          <span>{{item}}</span>
        </li>
      {{/collection}}
      <form class="form-inline">
        <input type="text">
        <input type="submit" value="Add" class="btn btn-primary">
      </form>
    """

  Application.setView new TodosView
